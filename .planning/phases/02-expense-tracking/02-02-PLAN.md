    model Settlement {
      settlementId String   @id @default(uuid()) @map("settlement_id")
      amount       Float
      createdAt    DateTime @default(now())

      toUserId     String @map("to_user_id")

      toUser   User      @relation("SettlementsTo", fields: [toUserId], references: [userId])
      household Household @relation(fields: [householdId], references: [householdId])
    }
